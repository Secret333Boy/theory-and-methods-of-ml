import numpy as np
import matplotlib.pyplot as plt

v = np.array([0, 1])
a = 1
M = np.matrix([[np.cos(a), -np.sin(a)], 
               [np.sin(a),  np.cos(a)]])
vs = [v]
for i in range(100):
    v = np.array(v * M * 0.99).flatten()
    vs.append(v)
    
vs = np.array(vs)

plt.figure(figsize=(6, 6))
plt.xlim(-1, 1)
plt.ylim(-1, 1)

plt.plot(*vs.T)
plt.show()

from math import pi

def rastrigin(x, y):
    return 20 + (np.pow(x, 2) - 10 * np.cos(2 * pi * x)) + (np.pow(y, 2) - 10 * np.cos(2 * pi * y))

x = np.linspace(-5, 5, 500)
y = np.linspace(-5, 5, 500)

[X, Y] = np.meshgrid(x,y)

Z = rastrigin(X, Y)

plt.figure(figsize=(10, 10))
plt.contourf(X, Y, Z, levels=10)
plt.show()
